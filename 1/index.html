TO obtain the transient response and time domain parameters for first and second order control systemss
s = %s;
T = syslin('c', 25, 25 + 4 * s + s^2);
t = 0:0.0005:5;
Ts = csim('s t e p', t, T);
plot2d(t, Ts);
xgrid;
xtitle('Response of II order function to unit step input for T(s)=25/(s^2+6s+25)', 'Time(sec)', 'c(t)');
y = denom(T);
z = coeff(y);
Wn = sqrt(z(1,1));
zeta = z(1,2) / (2 * Wn);
Wd = Wn * sqrt(1 - zeta^2);
Tp = %pi / Wd;
Mp = 100 * exp((-%pi * zeta) / sqrt(1 - zeta^2));
Td = (1 + 0.7 * zeta) / Wn;
a = atan(sqrt(1 - zeta^2) / zeta);
Tr = (%pi - a) / Wd;
Tset = 4 / (zeta * Wn);
peak_time = sprintf("Peak time = %6.3f secs", Tp);
peak_overshoot = sprintf("Peak overshoot = %6.3f percent", Mp);
Delay_time = sprintf("Delay time = %6.3f secs", Td);
Rise_time = sprintf("Rise time = %6.3f secs", Tr);
Setting_time = sprintf("Setting time = %6.3f secs", Tset);
