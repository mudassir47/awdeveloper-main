clear;clc;
xdel(winsid());
s = %s;
T = 1;
tf = syslin('c', 1 ,s^T+1);
t= 0 :0.001:25;
c = csim('imp',t,tf);
plot2d(t,c)
xgrid(5,1,7)
xtitle('impulse response','Time(sec)', 'c(t)')

clear;clc;
xdel(winsid());
s = %s;
T = 2;
G = syslin('c', 1 ,s^2+0.2*s+1);
t= 0 :0.5:50;
y = csim('imp',t,G);
plot(t,y)
xgrid(color('gray'))
xtitle('impulse response of ','Time(sec)', 'c(t)')
